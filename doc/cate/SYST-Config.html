模板解析引擎配置 


    SYST.tplConfig = {
        open: '{{',
        close: '}}'
    };

SYST.httpConfig Object

  • ajaxType 请求方式：get|post
  • ajaxDataType 返回的数据格式：xml, html, script, json, text, _default
  • commonData 所有请求中都需要传递的数据
  • ajaxBefore 发起请求之前执行
  • ajaxEnd 请求结束 success error 之前执行
  • ajaxComplete 请求完成 success error主题执行之后
  • ajaxSuccess 请求成功
  • ajaxError 请求失败

http 请求配置 



    var loadingGif;
    if(!$('#loading-gif')[0]){
        loadingGif = $('<img id="loading-gif" src="assets/images/loading.gif" style="display:none;" />');
        $('body').append(loadingGif);
    }

    SYST.httpConfig = {
        //-------------ajax请求过程的每一个回调回调（注意每一个请求都会调用）----------------------------------
        ajaxBefore: function(){
            //发起请求之前执行
            loadingGif.show();
        },
        ajaxEnd: function(res){
            //console.log('请求结束 success error 之前执行', res);
            loadingGif.hide();
        },
        ajaxComplete: function(res){
            //console.log('请求完成 success error主题执行之后', res);
        },
        ajaxSuccess: function(res){
            //console.log('请求成功', res);
        },
        ajaxError: function(err, xhr){
            //console.log('请求失败', this);
        }
    };
