var modelA = SYST.Model({
            _name_: 'Model A',
            age: 100,
            a: 'A'
        });

        var modelB = SYST.Model(modelA, {
            _name_: 'Model B',
            age: 101,
            b: 'B'
        });

        var modelC = SYST.Model(modelA, modelB, {
            _name_: 'Model C',
            age: 102,
            c: 'C'
        });
        console.log(modelB.a);
        console.log(modelC.a);
        console.log(modelC.b);
